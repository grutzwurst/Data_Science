\section{Matrixmultiplikation}
$
\left( 
	\begin{array}{rrr}
		4 & 2 & 3 \\
		5 & 1 & 6 \\
	\end{array}
 \right)
 \cdot
 \left(
	\begin{array}{rr}
		3 & 2 \\
		-1 & -3 \\
		6 & 8\\
	\end{array}
\right)$\\
$
=
\left(
	\begin{array}{cc}
		3\cdot4 - 1\cdot2 + 6\cdot3 & 2\cdot4 - 3\cdot2 + 8\cdot3 \\
		3\cdot5 - 1\cdot1 + 6\cdot6 & 2\cdot5 - 3\cdot1 + 8\cdot6 \\
	\end{array}
\right)$\\
$
=
\left(
	\begin{array}{cc}
		12-2+18 & 8-6+24 \\
		15-1+36 & 10-3+48\\
	\end{array}
\right)$\\
$
=
\left(
	\begin{array}{cc}
		28 & 26 \\
		50 & 55 \\
	\end{array}
\right)
$
